procedure OnLeftClick (aStr : String);
var
   Str : String;
   Race : Integer;
   nVirtue : Integer;
   CompleteQuest : Integer;
   CurrentQuest : Integer;
begin
   Str := callfunc ('getsenderrace');
   Race := StrToInt (Str);
   if Race <> 1 then begin
      exit;
   end;
   //제한된 사항들 체크.
   //1.호연지기체크
   //2.QuestNum 체크
   //3.아이템 체크
   Str := callfunc ('getsendervirtue');
   nVirtue := StrToInt (Str);
   if nVirtue < 4000 then begin
      print ('say 뵈횔꼇璃，供냥꼇죄훨蛟' );
      exit;
   end;
   
   Str := callfunc ('getsendercompletequest');
   CompleteQuest := StrToInt (Str);

   Str := callfunc ('getsendercurrentquest');
   CurrentQuest := StrToInt (Str);

   if CurrentQuest < 1250 then begin
      print ('say 첼莖떼唐봤懃헙');
      exit;
   end;
   
   if CurrentQuest = 1250 then begin
      Str := callfunc ('getsenderitemexistence 舅와寧뻔:1');
      if Str = 'false' then begin
         print ('changesendercompletequest 1000');
         print ('changesendercurrentquest 1000');
         CurrentQuest := 1000;
         CompleteQuest := 1000;
      end;
   end;

   if CurrentQuest = 1250 then begin
      Str := callfunc ('getsenderitemexistence 탬珙:1');
      if Str = 'true' then begin
         print ('say 델먕瞳乖충품찡큭嶠묘，꼇륩앎윱걸。。。');
         exit;
      end;
      
      Str := 'showwindow .\help\鮫堵침훙肯獗1.txt 0';
      print (Str);
      exit;
   end;

   if CurrentQuest > 1250 then begin
      if CurrentQuest < 1550 then begin
         print ('say 뻘청햄遼？속겉쐼랏율.');
         exit;
      end;
   end;
   
   if CurrentQuest = 1550 then begin
      Str := callfunc ('getsenderitemexistence 탬珙:1');
      if Str = 'false' then begin
         print ('changesendercompletequest 1200');
         print ('changesendercurrentquest 1250');
         CurrentQuest := 1250;
         CompleteQuest := 1200;
      end;

      Str := callfunc ('getsenderitemexistence 舅와寧뻔:1');
      if Str = 'false' then begin
         print ('changesendercompletequest 1000');
         print ('changesendercurrentquest 1000');
         CurrentQuest := 1000;
         CompleteQuest := 1000;
      end;

      print ('say 델먕瞳乖충품찡큭嶠묘，꼇륩앎윱걸。。。');
      exit;
   end;
end;

procedure OnGetResult (aStr : String);
var
   Str, Name : String;
begin
   if aStr = 'killok' then begin
      Str := callfunc ('checkenoughspace 2');
      if Str = 'false' then begin
         print ('say 膠틔으綠찮');
         exit;   
      end;

      print ('changesendercompletequest 1250');
      print ('changesendercurrentquest 1300');
      print ('putsendermagicitem 蝎변:1 @鮫堵침훙肯獗 4');
      print ('putsendermagicitem 鮫堵탬:1 @鮫堵침훙肯獗 4');
      print ('showwindow .\help\鮫堵침훙肯獗2.txt 0');
      exit;
   end;
   
   if aStr = 'q_close' then begin
      print ('say 맣긴寮雷죄，앎쀼윱');
      exit;
   end;

end;
